## 커스텀 컴포넌트

<InfoBox title="알림" type="info">
  이것은 MDX 안에서 사용되는 커스텀 컴포넌트입니다.
</InfoBox>

export function InfoBox({ title, type, children }) {
  const bgColor = type === 'info' ? 'bg-blue-100' : 'bg-gray-100'
  const textColor = type === 'info' ? 'text-blue-800' : 'text-gray-800'
  
  return (
    <div className={`p-4 rounded-lg ${bgColor} ${textColor} my-4`}>
      <h4 className="font-semibold mb-2">{title}</h4>
      <div>{children}</div>
    </div>
  )
}
